students = pd.read_csv("students.csv")
#display(students)
def assign_problems(
    students_df,
    problems_df,
    n_students,
    min_rank=None,
    max_rank=None,
    seed=None
):
    if seed is not None:
        random.seed(seed)

    # Filter problems by rank
    available_problems = problems_df.copy()

    if min_rank is not None:
        available_problems = available_problems[available_problems["rank"] >= min_rank]

    if max_rank is not None:
        available_problems = available_problems[available_problems["rank"] <= max_rank]

    # Check enough students
    if n_students > len(students_df):
        raise ValueError("You asked for more students than you have in the list.")

    # Check enough problems
    if n_students > len(available_problems):
        raise ValueError("Not enough problems in the selected rank range.")

    # Randomly choose students and problems
    chosen_students = students_df.sample(n=n_students, random_state=seed).reset_index(drop=True)
    chosen_problems = available_problems.sample(n=n_students, random_state=seed).reset_index(drop=True)

    # Combine
    result = pd.concat([chosen_students, chosen_problems], axis=1)

    return result
assignment = assign_problems(
    students_df=students,
    problems_df=problems,
    n_students=3,
    min_rank=1,
    max_rank=1,
    seed=None
)
